Method Description Example
push() Adds one or more elements to the end of an array and returns the new length. let arr = [1, 2];
arr.push(3); // [1, 2, 3]
pop() Removes the last element from an array and returns that element. let arr = [1, 2, 3];
arr.pop(); // [1, 2]
shift() Removes the first element from an array and returns that element. let arr = [1, 2, 3];
arr.shift(); // [2, 3]
unshift() Adds one or more elements to the beginning of an array and returns the new length. let arr = [2, 3];
arr.unshift(1); // [1, 2, 3]
filter() Creates a new array with all elements that pass the test provided by a function. let arr = [1, 2, 3];
arr.filter(n => n > 1); // [2, 3]
map() Creates a new array with the results of calling a function on every element. let arr = [1, 2, 3];
arr.map(n => n * 2); // [2, 4, 6]
reduce() Executes a reducer function on each element, resulting in a single output value. let arr = [1, 2, 3];
arr.reduce((sum, n) => sum + n, 0); // 6
find() Returns the value of the first element that satisfies the provided testing function. let arr = [1, 2, 3];
arr.find(n => n > 1); // 2
sort() Sorts the elements of an array in place and returns the sorted array. let arr = [3, 1, 2];
arr.sort(); // [1, 2, 3]
every() Tests whether all elements in the array pass the test implemented by a function. let arr = [1, 2, 3];
arr.every(n => n > 0); // true
some() Tests whether at least one element in the array passes the test implemented by a function. let arr = [1, 2, 3];
arr.some(n => n > 2); // true
forEach() Executes a provided function once for each array element. let arr = [1, 2, 3];
arr.forEach(n => console.log(n)); // logs 1, 2, 3
concat() Merges two or more arrays and returns a new array. let arr1 = [1, 2];
let arr2 = [3, 4];
arr1.concat(arr2); // [1, 2, 3, 4]
includes() Determines whether an array contains a certain element, returning true or false. let arr = [1, 2, 3];
arr.includes(2); // true
indexOf() Returns the first index at which a given element can be found, or -1 if it is not present. let arr = [1, 2, 3];
arr.indexOf(2); // 1
